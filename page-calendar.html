Single Datepicker

The Shared UI Library supports styling for the calendar, but to make it work you will need to include jQuery UI Datepicker.

Set an expiration date for your module.
Expiration date is empty or date inserted is not valid. Set an expiration date for your module.

HTML Markup

<div class="sui-form-field">

	<label for="unique-id" id="unique-id--label" class="sui-label">Label</label>

	<div class="sui-date">
		<input
			type="text"
			placeholder="Placeholder"
			id="unique-id"
			class="sui-form-control"
			aria-labelledby="unique-id--label"
			aria-describedby="unique-id--message unique-id--error"
		/>
		<span class="sui-icon-calendar" aria-hidden="true"></span>
	</div>

	<!-- NOTE: Print error message when error occurs and display message -->
	<span id="unique-id--error" class="sui-error-message" style="display: none;"></span>

	<span id="unique-id--message" class="sui-description">Field message goes here.</span>

</div>

Sample JS

$( '#unique-id' ).datepicker({
	beforeShow: function( input, inst ) {
		$( '#ui-datepicker-div' ).addClass( 'sui-calendar' );
	},
	'dateFormat': 'd MM yy'
});

Range Datepicker

For Range Datepicker our library supports styles but you will need to include latest jQuery, Moment.js and Date Range Picker Library.

Choose a date range to dynamically send your email.
Filling up automated email is required. Choose a date range to dynamically send your email.

<!-- Call: Moment.js -->
<script type="text/javascript" src="https://cdn.jsdelivr.net/momentjs/latest/moment.min.js"></script>

<!-- Call: Date Range Picker -->
<script type="text/javascript" src="https://cdn.jsdelivr.net/npm/daterangepicker/daterangepicker.min.js"></script>

Sample JS

( function( $ ) {

	$( 'body' ).ready( function() {

		let element = $( '#unique-id' ),
			start   = moment().subtract( 29, 'days' ),
			end     = moment();

		element.daterangepicker({
			'autoApply': true,
			ranges: {
				'Today': [ moment(), moment() ],
				'Tomorrow': [ moment().add( 1, 'days' ), moment().add( 1, 'days' ) ],
				'1 week': [ moment().startOf( 'week' ), moment().endOf( 'week' ) ],
				'30 days': [ moment().startOf( 'month' ), moment().endOf( 'month' ) ]
			},
			'locale': {
				'format': 'MMMM DD, YYYY',
				'separator': ' - ',
				'customRangeLabel': 'Custom',
				'daysOfWeek': [
					'Su',
					'Mo',
					'Tu',
					'We',
					'Th',
					'Fr',
					'Sa'
				],
				'monthNames': [
					'January',
					'February',
					'March',
					'April',
					'May',
					'June',
					'July',
					'August',
					'September',
					'October',
					'November',
					'December'
				]
			},
			'alwaysShowCalendars': true,
			'startDate': start,
			'endDate': end
		}, function( start, end, label ) {
			console.log( 'New date range selected: ' + start.format( 'YYYY-MM-DD' ) + ' to ' + end.format( 'YYYY-MM-DD' ) + ' (predefined range: ' + label + ')' );
		});

	});

}( jQuery ) );
