Documentation

Sample Markup

If you are using individual js components, tabs.js file is required. For more info go to tabs documentation.

<div class="sui-tabs sui-side-tabs">

	<div role="tablist" class="sui-tabs-menu">

		<!--
		NOTE(S):

		1. Include "type" attribute when using tabs inside a form to prevent
		problems when submitting the form.

		2. Include "aria-controls" attribute when using tabs with content.

		3. Unique IDs are a must for button tab and its content.
		-->

		<button
			type="button"
			role="tab"
			id="one__tab"
			class="sui-tab-item active"
			aria-controls="one__content"
			aria-selected="true">
			Tab 1
		</button>

		<button
			type="button"
			role="tab"
			id="two__tab"
			class="sui-tab-item"
			aria-controls="two__content"
			aria-selected="false"
			tabindex="-1">
			Tab 2
		</button>

	</div>

	<div class="sui-tabs-content">

		<div
			role="tabpanel"
			id="one__content"
			class="sui-tab-content active"
			aria-labelledby="one__tab"
			tabindex="0">

			<div class="sui-border-frame">
				<p>Tab 1</p>
			</div>

		</div>

		<div
			role="tabpanel"
			id="two__content"
			class="sui-tab-content"
			aria-labelledby="two__tab"
			tabindex="0"
			hidden>

			<div class="sui-border-frame">
				<p>Tab 2</p>
			</div>

		</div>

		<div
			role="tabpanel"
			id="sample-tab__three--content"
			class="sui-tab-content"
			aria-labelledby="sample-tab__three"
			tabindex="0"
			hidden>

			<div class="sui-border-frame">
				<p>Tab 3</p>
			</div>

		</div>

	</div>

</div>
